dim x as integer
dim y as integer
dim suit(4) as string
dim rank(13) as string
dim cards(4, 13) as string

suit(1) = "d"

suit(3) = "h"
suit(4) = "c"

rank(1) = "2"
rank(2) = "3"

rank(11) = "queen"
